CREATE TABLE `dictonary` (
  `ID` INT(11) NOT NULL AUTO_INCREMENT,
  `Word` LONGTEXT,
  `Translation` LONGTEXT,
  PRIMARY KEY (`ID`)
) ENGINE=INNODB DEFAULT CHARSET=latin1



CREATE TABLE `dictonarylanguage` (
  `ID` INT(11) NOT NULL AUTO_INCREMENT,
  `DictonaryID` INT(11) DEFAULT NULL,
  `LanguageID` INT(11) DEFAULT NULL,
  `Word` LONGTEXT,
  PRIMARY KEY (`ID`)
) ENGINE=INNODB DEFAULT CHARSET=latin1



CREATE TABLE `dictonarysource` (
  `ID` INT(11) NOT NULL AUTO_INCREMENT,
  `SourceID` INT(11) DEFAULT NULL,
  `Word` LONGTEXT,
  `Translation` LONGTEXT,
  PRIMARY KEY (`ID`)
) ENGINE=INNODB DEFAULT CHARSET=latin1
